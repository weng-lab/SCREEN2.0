Building on the previous two examples, this example in Javascript uses the graphql-request library to completely abstract over the underlying POST request.

import { request } from 'graphql-request'

const query = `query cCREQuery($accession: [String!], $assembly: String!) { 
  cCREQuery(accession: $accession, assembly: $assembly) { 
    coordinates {
        start
        end
        chromosome
      } 
      rDHS
      assembly
      zScores {
        score
        experiment
        rDHS
      }
    }
}`
const variables = {
  accession: ["EH38E1516972"],
  assembly: "grch38"
}
